SECTION B — GLOBAL RESPONSE FORMAT
B. GLOBAL RESPONSE FORMAT
Every backend function returns this exact structure. Frontend always checks success first.
GLOBAL RESPONSE STRUCTURE
-- Standard response wrapper applied to ALL RPC functions
{
  "success": boolean,   -- true = operation succeeded, false = it failed
  "message": string,    -- human-readable message for UI display
  "data":    {}         -- payload (null on errors)
}
-- SUCCESS example:
{
  "success": true,
  "message": "Order placed successfully",
  "data": {
    "order_id":    "a3f9-...",
    "total_amount": 45.00,
    "status":      "preparing"
  }
}
-- ERROR example:
{
  "success": false,
  "message": "OUT_OF_STOCK: Notebook has only 2 left (requested 5)",
  "data":    null
}
B.1 Wrapper Applied Inside All Key Functions
All key RPC functions wrap their output in the standard structure. The EXCEPTION block always returns success=false with the error
message.
B.1 — Wrapper Pattern
-- Pattern applied at the end of every function RETURN statement:
RETURN jsonb_build_object('success', true, 'message', 'Order placed', 'data',
    jsonb_build_object('order_id', v_order_id, 'total_amount', v_total));
-- Pattern applied in every EXCEPTION block:
PERFORM pg_notify('order_errors', jsonb_build_object('error', SQLERRM)::TEXT);
RAISE; -- Supabase converts RAISE to: {success:false, message:SQLERRM, data:null}
-- NOTE: When a PostgreSQL RAISE EXCEPTION fires, Supabase wraps it as:
-- { "code": "P0001", "message": "STORE_CLOSED: ...", "details": null }
-- Flutter catches this in the catch block and maps it to the standard format.
-- See Section C for the Flutter error-handling pattern.

C.3 Standard RPC Call Pattern (Flutter)
C.3 — Standard RPC Call Pattern
// STANDARD PATTERN for calling any backend function
Future<Map<String,dynamic>> callRpc(
    String function, Map<String,dynamic> params) async {
  try {
    final data = await supabase.rpc(function, params: params);
    // data is the JSONB returned by the PostgreSQL function
    return {
      'success': data['success'] ?? true,
      'message': data['message'] ?? 'OK',
      'data':    data['data']    ?? data,
    };
  } on PostgrestException catch (e) {
    // RAISE EXCEPTION in PostgreSQL becomes PostgrestException in Flutter
    return {
      'success': false,
      'message': e.message,  // e.g. 'STORE_CLOSED: Store is currently closed'
      'data':    null,
    };
  } catch (e) {
    return {'success': false, 'message': e.toString(), 'data': null};
  }

C.4 — Realtime Student Tracking
// Subscribe to order status changes — student tracking page
// When status changes to "ready", show token + time slot.
final channel = supabase
  .channel('order-tracking-' + orderId)
  .onPostgresChanges(
    event:  PostgresChangeEvent.update,
    schema: 'public',
    table:  'orders',
    filter: 'id=eq.' + orderId,
    callback: (payload) {
      final newStatus = payload.newRecord['status'];
      final token     = payload.newRecord['token_number'];
      final slotStart = payload.newRecord['slot_start'];
      final slotEnd   = payload.newRecord['slot_end'];
      if (newStatus == 'ready') {
        showOrderReadyScreen(token, slotStart, slotEnd);
      }
      if (newStatus == 'expired') {
        showRefundScreen();
      }
    })
  .subscribe();
// Unsubscribe when leaving page:
await supabase.removeChannel(channel);

CREATE OR REPLACE FUNCTION handle_new_user()
RETURNS TRIGGER LANGUAGE plpgsql SECURITY DEFINER SET search_path = public AS $$
BEGIN
    INSERT INTO profiles (id, full_name, role)
    VALUES (NEW.id, COALESCE(NEW.raw_user_meta_data->>'full_name',''), 'customer');
    RETURN NEW;
END; $$;
CREATE TRIGGER on_auth_user_created AFTER INSERT ON auth.users
    FOR EACH ROW EXECUTE FUNCTION handle_new_user();
-- Auto out-of-stock when stock reaches 0
CREATE OR REPLACE FUNCTION handle_stock_depletion()
RETURNS TRIGGER LANGUAGE plpgsql AS $$
BEGIN
    IF NEW.stock = 0 AND OLD.stock > 0 THEN NEW.is_available := false; END IF;
    RETURN NEW;
END; $$;
CREATE TRIGGER auto_out_of_stock BEFORE UPDATE OF stock ON items
    FOR EACH ROW EXECUTE FUNCTION handle_stock_depletion();

4. ORDER ENGINE — place_order
SECTION 4 — place_order
CREATE OR REPLACE FUNCTION revo.place_order(p_items JSONB, p_account TEXT DEFAULT NULL)
RETURNS JSONB LANGUAGE plpgsql SECURITY DEFINER SET search_path = public AS $$
DECLARE
    v_user_id    UUID; v_order_id UUID;
    v_item_ids   UUID[];
    v_total      NUMERIC(10,2) := 0; v_item_count INT := 0;
    v_item       RECORD; v_item_cache JSONB := jsonb_build_object(); v_cached JSONB;
    v_stock      INT; v_price NUMERIC(10,2); v_name TEXT; v_available BOOLEAN;
    v_is_open    BOOLEAN; v_limit INT; v_current INT;
    v_max_qty    INT; v_max_items INT; v_subtotal NUMERIC(10,2); v_affected INT;
BEGIN
    v_user_id := auth.uid();
    IF v_user_id IS NULL THEN RAISE EXCEPTION 'UNAUTHORIZED: No authenticated user'; END IF;
    IF p_items IS NULL OR jsonb_array_length(p_items) = 0 THEN RAISE EXCEPTION 'VALIDATION: Cart is empty'; END IF;

CREATE OR REPLACE FUNCTION revo.confirm_refund(p_order_id UUID, p_success BOOLEAN)
RETURNS JSONB LANGUAGE plpgsql SECURITY DEFINER SET search_path = public AS $$
DECLARE v_status refund_status;
BEGIN
    SELECT refund_status INTO v_status FROM orders WHERE id=p_order_id FOR UPDATE;
    IF v_status='refunded' THEN RETURN jsonb_build_object('success',true,'message','Already refunded','data',null); EN
D IF;
    IF v_status != 'processing' THEN RAISE EXCEPTION 'INVALID_STATE: refund_status is "%"', v_status; END IF;
    IF p_success THEN

    END IF;

CREATE OR REPLACE FUNCTION revo.retry_failed_refunds()
RETURNS INT LANGUAGE plpgsql SECURITY DEFINER SET search_path = public AS $$
DECLARE v_job RECORD; v_count INT:=0;
BEGIN
    FOR v_job IN SELECT id,payload,attempts,max_attempts FROM retry_queue
        WHERE job_type='refund' AND status='pending' AND next_retry_at<=now()
        ORDER BY created_at ASC LIMIT 50 FOR UPDATE SKIP LOCKED LOOP
        IF v_job.attempts >= v_job.max_attempts THEN
            UPDATE retry_queue SET status='dead',updated_at=now() WHERE id=v_job.id;
            PERFORM pg_notify('dead_letter',jsonb_build_object('job_id',v_job.id)::TEXT);
        ELSE
            UPDATE orders SET refund_status='pending',updated_at=now()
            WHERE id=(v_job.payload->>'order_id')::UUID AND refund_status='failed';
            UPDATE retry_queue SET attempts=attempts+1,status='processing',
                next_retry_at=now()+((2^v_job.attempts)||' minutes')::INTERVAL,updated_at=now() WHERE id=v_job.id;
            v_count:=v_count+1; END IF; END LOOP;
    RETURN v_count;
END; $$;

// supabase/functions/send-notification/index.ts
import { serve }        from 'https://deno.land/std@0.168.0/http/server.ts'
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'
serve(async (req) => {
  const payload = await req.json()
  const record  = payload?.record
  if (!record || record.status !== 'ready')
    return new Response('skip', { status: 200 })
  const supabase = createClient(
    Deno.env.get('SUPABASE_URL')!,
    Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!)
  const { data: profile } = await supabase
    .from('profiles').select('fcm_token').eq('id', record.user_id).single()
  if (!profile?.fcm_token) return new Response('no_token', { status: 200 })
  const fmt = (iso: string) => new Date(iso).toLocaleTimeString('en-IN',
    { hour: '2-digit', minute: '2-digit', hour12: true })
  const dateStr = new Date(record.slot_start).toLocaleDateString('en-IN',
    { weekday: 'long', day: 'numeric', month: 'long', year: 'numeric' })
  await fetch(
    `https://fcm.googleapis.com/v1/projects/${Deno.env.get('FCM_PROJECT_ID')}/messages:send`,
    { method: "POST",
      headers: { 'Authorization': `Bearer ${Deno.env.get('FCM_SERVER_KEY')}`, 'Content-Type': 'application/json' },
      body: JSON.stringify({ message: {
        token: profile.fcm_token,
        notification: { title: 'Your order is ready!',
          body: `Token #${record.token_number} | Collect: ${fmt(record.slot_start)} to ${fmt(record.slot_end)}` },
        data: { type:'ORDER_READY', order_id:record.id,
               token_number:String(record.token_number),
               slot_start:record.slot_start, slot_end:record.slot_end, date:dateStr }}})})
  return new Response('sent', { status: 200 })
})

CREATE OR REPLACE FUNCTION revo.confirm_payment(p_order_id UUID, p_gateway_id TEXT)
RETURNS JSONB LANGUAGE plpgsql SECURITY DEFINER SET search_path = public AS $$
DECLARE v_user_id UUID; v_order_uid UUID; v_current_pay payment_status; v_total NUMERIC(10,2);
BEGIN
    v_user_id:=auth.uid(); IF v_user_id IS NULL THEN RAISE EXCEPTION 'UNAUTHORIZED'; END IF;
    SELECT user_id,payment_status,total_amount INTO v_order_uid,v_current_pay,v_total
    FROM orders WHERE id=p_order_id FOR UPDATE;
    IF v_order_uid IS NULL THEN RAISE EXCEPTION 'NOT_FOUND'; END IF;
    IF v_order_uid!=v_user_id THEN RAISE EXCEPTION 'FORBIDDEN'; END IF;
    IF v_current_pay='paid' THEN RETURN jsonb_build_object('success',true,'message','Already paid','data',null); END I
F;
    IF v_current_pay!='pending' THEN RAISE EXCEPTION 'INVALID_STATE: %',v_current_pay; END IF;
    UPDATE orders SET payment_status='paid',payment_gateway_id=p_gateway_id,payment_confirmed_at=now(),updated_at=now(
) WHERE id=p_order_id;
    RETURN jsonb_build_object('success',true,'message','Payment confirmed','data',
        jsonb_build_object('order_id',p_order_id,'total_amount',v_total,'gateway_id',p_gateway_id));
END; $$;
GRANT EXECUTE ON FUNCTION revo.confirm_payment TO authenticated;

// supabase/functions/razorpay-webhook/index.ts
import { serve }       from 'https://deno.land/std@0.168.0/http/server.ts'
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'
import { createHmac }  from 'https://deno.land/std@0.168.0/node/crypto.ts'
serve(async (req) => {
  const body=await req.text(); const sig=req.headers.get("x-razorpay-signature")??"";
  const secret=Deno.env.get('RAZORPAY_WEBHOOK_SECRET')!;
  const expected=createHmac("sha256",secret).update(body).digest("hex");
  if (expected!==sig) return new Response('Invalid signature',{status:401});
  const event=JSON.parse(body);
  if (event.event!=='payment.captured') return new Response('ignored',{status:200});
  const supabase=createClient(Deno.env.get('SUPABASE_URL')!,Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!);
  const p=event.payload.payment.entity;
  await supabase.rpc('confirm_payment',{p_order_id:p.notes.order_id,p_gateway_id:p.id});
  return new Response('OK',{status:200});
})
